1//! # Signal Generation and Analysis Module
2//!
3//! This module provides comprehensive signal generation and analysis capabilities
4//! for power electronics applications. It supports various waveform types commonly
5//! encountered in power systems and includes detailed signal metrics computation.
6//!
7//! ## Supported Signal Types
8//!
9//! - **DC**: Constant voltage/current signals
10//! - **Sine**: Fundamental AC waveforms
11//! - **Full-wave rectified sine**: Used in DC power supplies
12//! - **Half-wave rectified sine**: Simple rectifier outputs
13//! - **PWM**: Pulse Width Modulation signals for switching converters
14//! - **Triangular**: Carrier waves for PWM generation
15//! - **Square**: Digital control signals and switching waveforms
16//! - **Sawtooth**: Linear ramp signals for timing and control
17//!
18//! ## Signal Metrics
19//!
20//! The module computes comprehensive signal quality metrics:
21//! - Peak-to-peak, maximum, minimum values
22//! - Average (DC component) and RMS values
23//! - Duty cycle analysis (positive and negative)
24//! - Rise and fall time measurements
25//! - Crest factor and THD (Total Harmonic Distortion)
26//!
27//! ## Applications
28//!
29//! - Power quality analysis
30//! - Converter waveform generation
31//! - System testing and validation
32//! - Control signal synthesis
33//! - Harmonic analysis
34

39/// Signal type enumeration for waveform generation.
40///
41/// Each variant represents a different mathematical function used
42/// to generate time-domain waveforms commonly found in power electronics.
43#[derive(Debug, Clone, PartialEq)]
44pub enum SignalType {
45    /// Constant DC signal: f(t) = amplitude
46    DC,
47    /// Sinusoidal signal: f(t) = amplitude × sin(2πft + φ)
48    Sine,
49    /// Full-wave rectified sine: f(t) = amplitude × |sin(2πft + φ)|
50    FullWaveRectifiedSine,
51    /// Half-wave rectified sine: f(t) = amplitude × max(0, sin(2πft + φ))
52    HalfWaveRectifiedSine,
53    /// PWM signal with configurable duty cycle
54    PWM,
55    /// Triangular wave: Linear ramp up and down
56    Triangular,
57    /// Square wave: Alternating high and low levels
58    Square,
59    /// Sawtooth wave: Linear ramp with sharp reset
60    Sawtooth,
61}

63/// Signal generator and analyzer for power electronics waveforms.
64///
65/// This struct provides both signal generation capabilities and comprehensive
66/// signal analysis including quality metrics computation. It operates on
67/// a provided sample buffer to maintain `no_std` compatibility.
68///
69/// # Examples
70///
71/// ## Sine Wave Generation
72///
73/// ```rust
74/// use libpower::signal::signal::{Signal, SignalType};
75///
76/// let mut buffer = [0.0; 1000];
77/// let mut signal = Signal::new(SignalType::Sine, 10.0, 50.0, &mut buffer);
78/// signal.set_phase(0.0);
79///
80/// let samples = signal.get_samples();
81/// let metrics = signal.get_metrics();
82///
83/// // Verify sine wave properties
84/// assert!((metrics.dc_rms - 7.07).abs() < 0.1); // ~10/√2
85/// assert!(metrics.average.abs() < 0.1); // Zero average for pure sine
86/// ```
87///
88/// ## PWM Signal Analysis
89///
90/// ```rust
91/// use libpower::signal::signal::{Signal, SignalType};
92///
93/// let mut buffer = [0.0; 1000];
94/// let mut signal = Signal::new(SignalType::PWM, 5.0, 1000.0, &mut buffer);
95/// signal.set_duty_cycle(0.25); // 25% duty cycle
96///
97/// let metrics = signal.get_metrics();
98/// assert!((metrics.duty_cycle_pos - 0.25).abs() < 0.01);
99/// assert!((metrics.average - 1.25).abs() < 0.1); // 25% of 5V
100/// ```
101///
102/// ## Signal Quality Metrics
103///
104/// ```rust
105/// use libpower::signal::signal::{Signal, SignalType};
106///
107/// let mut buffer = [0.0; 500];
108/// let mut signal = Signal::new(SignalType::Square, 12.0, 60.0, &mut buffer);
109///
110/// let metrics = signal.get_metrics();
111/// assert_eq!(metrics.max, 12.0);
112/// assert_eq!(metrics.min, -12.0);
113/// assert_eq!(metrics.peak_to_peak, 24.0);
114/// assert!((metrics.crest_factor - 1.0).abs() < 0.1); // Square wave has crest factor ≈ 1.0
115/// ```
116pub struct Signal<'a> {
117    /// Type of waveform to generate
118    wave_type: SignalType,
119    /// Peak amplitude of the signal
120    amplitude: f32,
121    /// Phase offset in radians
122    phase: f32,
123    /// Fundamental frequency in Hz
124    frequency: f32,
125    /// Number of samples in the buffer
126    num_samples: u32,
127    /// Duty cycle for PWM signals (0.0 to 1.0)
128    duty_cycle: f32,
129    // Internal measurements
130    /// Peak-to-peak amplitude
131    peak_to_peak: f32,
132    /// Maximum sample value
133    max: f32,
134    /// Minimum sample value
135    min: f32,
136    /// Average (DC component)
137    average: f32,
138    /// DC RMS value
139    dc_rms: f32,
140    /// AC RMS value
141    ac_rms: f32,
142    /// Positive duty cycle
143    duty_cycle_pos: f32,
144    /// Negative duty cycle
145    duty_cycle_neg: f32,
146    /// Rise time measurement
147    rise_time: f32,
148    /// Fall time measurement
149    fall_time: f32,
150    /// Crest factor (peak/RMS)
151    crest_factor: f32,
152    /// Total Harmonic Distortion
153    thd: f32,
154    /// Sample buffer reference
155    samples: &'a mut [f32], // Use slice instead of Vec for no_std
156}

158/// Comprehensive signal quality metrics.
159///
160/// This structure contains all computed signal characteristics useful
161/// for power electronics analysis and system characterization.
162#[derive(Debug, Clone, Copy)]
163pub struct SignalMetrics {
164    /// Peak-to-peak amplitude (max - min)
165    pub peak_to_peak: f32,
166    /// Maximum sample value
167    pub max: f32,
168    /// Minimum sample value
169    pub min: f32,
170    /// Average value (DC component)
171    pub average: f32,
172    /// DC RMS value
173    pub dc_rms: f32,
174    /// AC RMS value (excluding DC)
175    pub ac_rms: f32,
176    /// Positive duty cycle (fraction of time above zero)
177    pub duty_cycle_pos: f32,
178    /// Negative duty cycle (fraction of time below zero)
179    pub duty_cycle_neg: f32,
180    /// Rise time (10% to 90% of amplitude)
181    pub rise_time: f32,
182    /// Fall time (90% to 10% of amplitude)
183    pub fall_time: f32,
184    /// Crest factor (peak amplitude / RMS)
185    pub crest_factor: f32,
186    /// Total Harmonic Distortion percentage
187    pub thd: f32,
188}

216        frequency: f32,
217        samples_buffer: &'a mut [f32],
218    ) -> Signal<'a> {

220        let mut signal = Signal {
221            wave_type,

223            phase: 0.0,

225            num_samples,
226            duty_cycle: 0.5, // Default 50% duty cycle
227            peak_to_peak: 0.0,
228            max: 0.0,
229            min: 0.0,
230            average: 0.0,
231            dc_rms: 0.0,
232            ac_rms: 0.0,
233            duty_cycle_pos: 0.0,
234            duty_cycle_neg: 0.0,
235            rise_time: 0.0,
236            fall_time: 0.0,
237            crest_factor: 0.0,
238            thd: 0.0,
239            samples: samples_buffer,
240        };
241
242        signal.generate_samples();
243        signal.calculate_metrics();
244        signal

320    /// Generates signal samples based on the current parameters.
321    ///
322    /// This method fills the sample buffer with the appropriate waveform
323    /// based on the signal type, amplitude, frequency, and other parameters.
324    fn generate_samples(&mut self) {
325        let sample_rate = self.num_samples as f32;
326
327        for i in 0..self.num_samples {
328            let t = i as f32 / sample_rate;
329            let angular_freq = 2.0 * PI * self.frequency;
330            let phase_rad = self.phase * PI / 180.0;
331
332            let sample = match self.wave_type {
333                SignalType::DC => self.amplitude,
334
335                SignalType::Sine => self.amplitude * libm::sinf(angular_freq * t + phase_rad),
336
337                SignalType::FullWaveRectifiedSine => {
338                    self.amplitude * libm::fabsf(libm::sinf(angular_freq * t + phase_rad))
339                }
340
341                SignalType::HalfWaveRectifiedSine => {
342                    let val = self.amplitude * libm::sinf(angular_freq * t + phase_rad);
343                    if val > 0.0 { val } else { 0.0 }
344                }
345
346                SignalType::PWM => {
347                    // For PWM, we assume the buffer represents one or more complete periods
348                    // If frequency is high relative to num_samples, we just create a simple pattern
349                    let samples_per_period = if self.frequency >= self.num_samples as f32 {
350                        // High frequency case: treat the whole buffer as one period
351                        self.num_samples as f32
352                    } else {
353                        // Normal case: calculate samples per period
354                        self.num_samples as f32 / self.frequency
355                    };
356
357                    let sample_in_period = (i as f32) % samples_per_period;
358                    let normalized_position = sample_in_period / samples_per_period;
359
360                    if normalized_position < self.duty_cycle {
361                        self.amplitude
362                    } else {
363                        0.0
364                    }
365                }
366
367                SignalType::Triangular => {
368                    let period = 1.0 / self.frequency;
369                    let t_mod = (t + phase_rad / angular_freq) % period;
370                    let normalized = t_mod / period;
371
372                    if normalized < 0.5 {
373                        self.amplitude * (4.0 * normalized - 1.0)
374                    } else {
375                        self.amplitude * (3.0 - 4.0 * normalized)
376                    }
377                }
378
379                SignalType::Square => {
380                    if libm::sinf(angular_freq * t + phase_rad) >= 0.0 {
381                        self.amplitude
382                    } else {
383                        -self.amplitude
384                    }
385                }
386
387                SignalType::Sawtooth => {
388                    let period = 1.0 / self.frequency;
389                    let t_mod = (t + phase_rad / angular_freq) % period;
390                    self.amplitude * (2.0 * t_mod / period - 1.0)
391                }
392            };
393
394            self.samples[i as usize] = sample;
395        }
396    }

398    /// Calculates comprehensive signal metrics from the generated samples.
399    ///
400    /// This method computes all signal quality metrics including statistical
401    /// measures, duty cycles, timing parameters, and harmonic content.
402    fn calculate_metrics(&mut self) {
403        // Basic statistics
404        self.max = f32::NEG_INFINITY;
405        self.min = f32::INFINITY;
406        let mut sum = 0.0;
407        let mut squared_sum = 0.0;
408        let mut positive_samples = 0;
409
410        for &sample in self.samples.iter() {
411            self.max = libm::fmaxf(self.max, sample);
412            self.min = libm::fminf(self.min, sample);
413            sum += sample;
414            squared_sum += sample * sample;
415            if sample > 0.0 {
416                positive_samples += 1;
417            }
418        }
419
420        self.peak_to_peak = self.max - self.min;
421        self.average = sum / self.num_samples as f32;
422
423        // RMS calculations
424        self.dc_rms = libm::sqrtf(squared_sum / self.num_samples as f32);
425
426        // AC RMS (remove DC component)
427        let mut ac_squared_sum = 0.0;
428        for &sample in self.samples.iter() {
429            let ac_sample = sample - self.average;
430            ac_squared_sum += ac_sample * ac_sample;
431        }
432        self.ac_rms = libm::sqrtf(ac_squared_sum / self.num_samples as f32);
433
434        // Duty cycle calculations
435        if self.wave_type == SignalType::PWM {
436            // For PWM, count samples at high level (amplitude)
437            let mut high_samples = 0;
438            for &sample in self.samples.iter() {
439                if (sample - self.amplitude).abs() < 1e-6 {
440                    high_samples += 1;
441                }
442            }
443            self.duty_cycle_pos = high_samples as f32 / self.num_samples as f32;
444            self.duty_cycle_neg = 1.0 - self.duty_cycle_pos;
445        } else {
446            // For other signals, count positive samples
447            self.duty_cycle_pos = positive_samples as f32 / self.num_samples as f32;
448            self.duty_cycle_neg = 1.0 - self.duty_cycle_pos;
449        }
450
451        // Rise and fall time calculations (10% to 90%)
452        let threshold_low = self.min + 0.1 * self.peak_to_peak;
453        let threshold_high = self.min + 0.9 * self.peak_to_peak;
454
455        let mut rise_samples = 0;
456        let mut fall_samples = 0;
457        let mut rising = false;
458        let mut falling = false;
459
460        for i in 1..self.samples.len() {
461            let prev = self.samples[i - 1];
462            let curr = self.samples[i];
463
464            if prev <= threshold_low && curr > threshold_low {
465                rising = true;
466                falling = false;
467                rise_samples = 0;
468            }
469            if rising && curr <= threshold_high {
470                rise_samples += 1;
471            }
472            if prev >= threshold_high && curr < threshold_high {
473                falling = true;
474                rising = false;
475                fall_samples = 0;
476            }
477            if falling && curr >= threshold_low {
478                fall_samples += 1;
479            }
480        }
481
482        let sample_period = 1.0 / (self.frequency * self.num_samples as f32);
483        self.rise_time = rise_samples as f32 * sample_period;
484        self.fall_time = fall_samples as f32 * sample_period;
485
486        // Crest factor: peak value divided by RMS value
487        if self.dc_rms != 0.0 {
488            let peak = libm::fmaxf(libm::fabsf(self.max), libm::fabsf(self.min));
489            self.crest_factor = peak / self.dc_rms;
490        }
491
492        // Total Harmonic Distortion (simplified - only considers first 5 harmonics)
493        if self.wave_type == SignalType::Sine {
494            let fundamental = self.frequency;
495            let mut harmonic_power = 0.0;
496
497            for i in 2..=5 {
498                let harmonic_freq = fundamental * i as f32;
499                let harmonic_amplitude = self.compute_fft_magnitude(harmonic_freq);
500                harmonic_power += harmonic_amplitude * harmonic_amplitude;
501            }
502
503            let fundamental_amplitude = self.compute_fft_magnitude(fundamental);
504            if fundamental_amplitude != 0.0 {
505                self.thd = (libm::sqrtf(harmonic_power) / fundamental_amplitude) * 100.0;
506            }
507        }
508    }

510    /// Computes simplified FFT magnitude for THD estimation.
511    ///
512    /// This is a simplified implementation for demonstration purposes.
513    /// A full THD calculation would require a complete FFT implementation.
514    ///
515    /// # Parameters
516    ///
517    /// * `frequency` - Fundamental frequency for analysis
518    ///
519    /// # Returns
520    ///
521    /// Estimated THD percentage as a simplified metric.
522    fn compute_fft_magnitude(&self, frequency: f32) -> f32 {
523        let sample_rate = self.num_samples as f32;
524        let mut real = 0.0;
525        let mut imag = 0.0;
526
527        for (i, &sample) in self.samples.iter().enumerate() {
528            let t = i as f32 / sample_rate;
529            let angle = 2.0 * PI * frequency * t;
530            real += sample * libm::cosf(angle);
531            imag += sample * libm::sinf(angle);
532        }
533
534        libm::sqrtf(real * real + imag * imag) / self.num_samples as f32
535    }
