1//! # Clarke Transformation (abc → αβ0)
2//!
3//! This module implements the Clarke transformation, which converts three-phase
4//! quantities (abc coordinates) into a two-phase orthogonal stationary reference
5//! frame (αβ coordinates) plus a zero-sequence component. This transformation
6//! is fundamental in three-phase power electronics and motor control applications.
7//!
8//! ## Theory
9//!
10//! The Clarke transformation converts three symmetrical AC quantities into two
11//! orthogonal components plus a zero-sequence component:
12//!
13//! ### Mathematical Definition
14//!
15//! **Forward Clarke Transform (abc → αβ0):**
16//! ```text
17//! [α]   [1     -1/2    -1/2 ] [a]
18//! [β] = [0   √3/2   -√3/2 ] [b] × (2/3)
19//! [0]   [1/2   1/2     1/2 ] [c]
20//! ```
21//!
22//! ### Expanded Form
23//!
24//! ```text
25//! α = (2/3) × (a - b/2 - c/2)      // Aligned with phase A
26//! β = (2/3) × (√3/2) × (b - c)     // Orthogonal to α (90° ahead)
27//! 0 = (1/3) × (a + b + c)          // Zero-sequence (unbalanced component)
28//! ```
29//!
30//! ## Power Invariance
31//!
32//! The transformation preserves instantaneous power:
33//! ```text
34//! Pa + Pb + Pc = (3/2)(Pα + Pβ) + 3×P0
35//! ```
36//!
37//! ## Physical Interpretation
38//!
39//! - **α-axis**: Aligned with phase A axis
40//! - **β-axis**: 90° ahead of α-axis in electrical degrees
41//! - **0-sequence**: Common mode component (indicates unbalance)
42//!
43//! ## Applications
44//!
45//! - **Vector control of AC motors**: Foundation for field-oriented control
46//! - **Grid-tied inverters**: Converting three-phase measurements to αβ frame
47//! - **Power quality analysis**: Decomposing three-phase unbalanced quantities
48//! - **Space vector PWM**: Generating optimal switching patterns
49//! - **Active power filters**: Harmonic analysis and compensation
50//! - **Wind turbine control**: Grid synchronization and power control
51

54/// Three-phase to two-phase Clarke transformation.
55///
56/// This structure provides both stateful and stateless Clarke transformation
57/// operations. The stateful version maintains internal state for repeated
58/// calculations, while the stateless version provides utility functions
59/// for one-time calculations.
60///
61/// # Examples
62///
63/// ## Stateful Operation
64///
65/// ```rust
66/// use libpower::transform::clarke::Clarke;
67///
68/// // Create transformer and set inputs
69/// let mut clarke = Clarke::new(0.0, 0.0);
70/// clarke.set_inputs(100.0, -50.0, -50.0);  // Balanced three-phase
71///
72/// // Perform transformation  
73/// clarke.calculate_stateful();
74///
75/// // Access results
76/// let alpha = clarke.get_alpha();
77/// let beta = clarke.get_beta();
78/// let zero = clarke.get_zero();
79///
80/// println!("α: {:.2}, β: {:.2}, 0: {:.2}", alpha, beta, zero);
81/// assert!(zero.abs() < 1e-6);  // Should be balanced (zero ≈ 0)
82/// ```
83///
84/// ## Stateless Operation
85///
86/// ```rust
87/// use libpower::transform::clarke::Clarke;
88///
89/// let clarke = Clarke::new(0.0, 0.0);
90/// let result = clarke.calculate(100.0, -50.0, -50.0);
91///
92/// println!("α: {:.2}, β: {:.2}, 0: {:.2}",
93///          result.alpha, result.beta, result.zero);
94/// ```
95///
96/// ## Motor Control Application
97///
98/// ```rust
99/// use libpower::transform::clarke::Clarke;
100///
101/// let mut clarke = Clarke::new(0.0, 0.0);
102///
103/// // Three-phase current measurements
104/// let ia = 10.0;   // Phase A current
105/// let ib = -5.0;   // Phase B current  
106/// let ic = -5.0;   // Phase C current
107///
108/// clarke.set_inputs(ia, ib, ic);
109/// clarke.calculate_stateful();
110///
111/// // Use α-β currents for space vector control
112/// let i_alpha = clarke.get_alpha();
113/// let i_beta = clarke.get_beta();
114///
115/// // Check if system is balanced
116/// let zero_seq = clarke.get_zero();
117/// let is_balanced = zero_seq.abs() < 1.0;  // 1A tolerance
118///
119/// assert!(is_balanced);
120/// ```
121#[derive(Debug, Clone)]
122pub struct Clarke {
123    /// Phase A input value
124    a: f32,
125    /// Phase B input value
126    b: f32,
127    /// Phase C input value
128    c: f32,
129    /// Alpha component output (α)
130    alpha: f32,
131    /// Beta component output (β)
132    beta: f32,
133    /// Zero-sequence component output (0)
134    zero: f32,
135}

320    /// Performs the stateful Clarke transformation calculation.
321    ///
322    /// This method computes the Clarke transformation using the internally stored
323    /// abc values (set via [`set_inputs`](#method.set_inputs)) and updates the
324    /// internal α, β, and zero-sequence components.
325    ///
326    /// # Mathematical Implementation
327    ///
328    /// Uses the power-invariant Clarke transformation:
329    /// ```text
330    /// α = (2/3) × (a - b/2 - c/2)
331    /// β = (2/3) × (√3/2) × (b - c)  
332    /// 0 = (1/3) × (a + b + c)
333    /// ```
334    ///
335    /// # Examples
336    ///
337    /// ## Balanced Three-Phase System
338    ///
339    /// ```rust
340    /// use libpower::transform::clarke::Clarke;
341    ///
342    /// let mut clarke = Clarke::new(0.0, 0.0);
343    ///
344    /// // 120° phase-shifted sinusoidal system
345    /// clarke.set_inputs(100.0, -50.0, -50.0);
346    /// clarke.calculate_stateful();
347    ///
348    /// let alpha = clarke.get_alpha();
349    /// let beta = clarke.get_beta();
350    /// let zero = clarke.get_zero();
351    ///
352    /// // Balanced system: zero-sequence ≈ 0
353    /// assert!(zero.abs() < 1e-6);
354    ///
355    /// // Check magnitude preservation (approximately)
356    /// let abc_rms = (100.0_f32.powi(2) + 50.0_f32.powi(2) + 50.0_f32.powi(2)).sqrt() / 3.0_f32.sqrt();
357    /// let ab_rms = (alpha.powi(2) + beta.powi(2)).sqrt() / (3.0_f32/2.0).sqrt();
358    /// ```
359    ///
360    /// ## Unbalanced System Detection
361    ///
362    /// ```rust
363    /// use libpower::transform::clarke::Clarke;
364    ///
365    /// let mut clarke = Clarke::new(0.0, 0.0);
366    ///
367    /// // Unbalanced system with DC offset
368    /// clarke.set_inputs(110.0, -40.0, -50.0);  // Sum ≠ 0
369    /// clarke.calculate_stateful();
370    ///
371    /// let zero_seq = clarke.get_zero();
372    /// assert!(zero_seq.abs() > 5.0);  // Significant unbalance
373    ///
374    /// println!("Unbalance detected: {:.2}", zero_seq);
375    /// ```
376    pub fn calculate_stateful(&mut self) {
377        // Power-invariant Clarke transformation
378        // Constants for transformation
379        const TWO_THIRDS: f32 = 2.0 / 3.0;
380        const SQRT3_OVER_2: f32 = 0.8660254037844387; // √3/2
381        const ONE_THIRD: f32 = 1.0 / 3.0;
382
383        // Calculate α component: (2/3) * (a - b/2 - c/2)
384        self.alpha = TWO_THIRDS * (self.a - 0.5 * self.b - 0.5 * self.c);
385
386        // Calculate β component: (2/3) * (√3/2) * (b - c)
387        self.beta = TWO_THIRDS * SQRT3_OVER_2 * (self.b - self.c);
388
389        // Calculate zero-sequence component: (1/3) * (a + b + c)
390        self.zero = ONE_THIRD * (self.a + self.b + self.c);
391    }

393    /// Performs Clarke transformation on given abc values (stateless).
394    ///
395    /// This method computes the Clarke transformation for the provided abc inputs
396    /// without modifying the internal state. It returns a `ClarkeResult` structure
397    /// containing the calculated components.
398    ///
399    /// # Parameters
400    ///
401    /// * `a` - Phase A quantity
402    /// * `b` - Phase B quantity
403    /// * `c` - Phase C quantity
404    ///
405    /// # Returns
406    ///
407    /// A `ClarkeResult` containing the α, β, and zero-sequence components.
408    ///
409    /// # Examples
410    ///
411    /// ## Basic Transformation
412    ///
413    /// ```rust
414    /// use libpower::transform::clarke::Clarke;
415    ///
416    /// let clarke = Clarke::new(0.0, 0.0);
417    /// let result = clarke.calculate(100.0, -50.0, -50.0);
418    ///
419    /// println!("α: {:.2}, β: {:.2}, 0: {:.2}",
420    ///          result.alpha, result.beta, result.zero);
421    ///
422    /// // Verify balanced system
423    /// assert!(result.zero.abs() < 1e-6);
424    /// ```
425    ///
426    /// ## Real-time Processing
427    ///
428    /// ```rust
429    /// use libpower::transform::clarke::Clarke;
430    /// use std::f32::consts::PI;
431    ///
432    /// let clarke = Clarke::new(0.0, 0.0);
433    /// let mut results = Vec::new();
434    ///
435    /// // Process samples from three-phase system
436    /// for i in 0..10 {
437    ///     let t = i as f32 * 0.001; // 1ms time steps
438    ///     let omega = 2.0 * PI * 50.0; // 50 Hz
439    ///     
440    ///     // Generate three-phase sinusoidal signals
441    ///     let v_a = 325.0 * (omega * t).sin();
442    ///     let v_b = 325.0 * (omega * t - 2.0 * PI / 3.0).sin();
443    ///     let v_c = 325.0 * (omega * t + 2.0 * PI / 3.0).sin();
444    ///     
445    ///     let clarke_result = clarke.calculate(v_a, v_b, v_c);
446    ///     results.push(clarke_result);
447    /// }
448    ///
449    /// // Process results for control algorithm
450    /// for result in results {
451    ///     // Use α-β components for space vector control
452    ///     let magnitude = (result.alpha.powi(2) + result.beta.powi(2)).sqrt();
453    ///     let angle = clarke.angle(&result);
454    /// }
455    /// ```
456    pub fn calculate(&self, a: f32, b: f32, c: f32) -> ClarkeResult {
457        // Power-invariant Clarke transformation
458        // Constants for transformation
459        const TWO_THIRDS: f32 = 2.0 / 3.0;
460        const SQRT3_OVER_2: f32 = 0.8660254037844387; // √3/2
461        const ONE_THIRD: f32 = 1.0 / 3.0;
462
463        // Calculate α component: (2/3) * (a - b/2 - c/2)
464        let alpha = TWO_THIRDS * (a - 0.5 * b - 0.5 * c);
465
466        // Calculate β component: (2/3) * (√3/2) * (b - c)
467        let beta = TWO_THIRDS * SQRT3_OVER_2 * (b - c);
468
469        // Calculate zero-sequence component: (1/3) * (a + b + c)
470        let zero = ONE_THIRD * (a + b + c);
471
472        ClarkeResult { alpha, beta, zero }
473    }

475    /// Performs Clarke transformation using amplitude-invariant form.
476    ///
477    /// This variant preserves the amplitude of the original signals rather than power.
478    /// The peak values of α and β components equal the peak values of the abc components.
479    ///
480    /// # Parameters
481    ///
482    /// * `a` - Phase A quantity
483    /// * `b` - Phase B quantity  
484    /// * `c` - Phase C quantity
485    ///
486    /// # Returns
487    ///
488    /// A `ClarkeResult` with amplitude-invariant α, β, and zero components.
489    ///
490    /// # Mathematical Implementation
491    ///
492    /// ```text
493    /// α = a - b/2 - c/2
494    /// β = (√3/2) × (b - c)
495    /// 0 = (1/3) × (a + b + c)
496    /// ```
497    ///
498    /// # Examples
499    ///
500    /// ```rust
501    /// use libpower::transform::clarke::Clarke;
502    ///
503    /// let clarke = Clarke::new(0.0, 0.0);
504    ///
505    /// // Compare power-invariant vs amplitude-invariant
506    /// let (a, b, c) = (100.0, -50.0, -50.0);
507    ///
508    /// let power_inv = clarke.calculate(a, b, c);
509    /// let amp_inv = clarke.calculate_amplitude_invariant(a, b, c);
510    ///
511    /// // Amplitude-invariant gives larger values
512    /// assert!(amp_inv.alpha > power_inv.alpha);
513    /// println!("Power-inv α: {:.1}, Amp-inv α: {:.1}",
514    ///          power_inv.alpha, amp_inv.alpha);
515    /// ```
516    pub fn calculate_amplitude_invariant(&self, a: f32, b: f32, c: f32) -> ClarkeResult {
517        // Amplitude-invariant Clarke transformation
518        const SQRT3_OVER_2: f32 = 0.8660254037844387; // √3/2
519        const ONE_THIRD: f32 = 1.0 / 3.0;
520
521        // Calculate α component: a - b/2 - c/2
522        let alpha = a - 0.5 * b - 0.5 * c;
523
524        // Calculate β component: (√3/2) * (b - c)
525        let beta = SQRT3_OVER_2 * (b - c);
526
527        // Calculate zero-sequence component: (1/3) * (a + b + c)
528        let zero = ONE_THIRD * (a + b + c);
529
530        ClarkeResult { alpha, beta, zero }
531    }

533    /// Calculates the magnitude of the αβ vector.
534    ///
535    /// Computes the magnitude of the space vector represented by the α and β components.
536    /// This is useful for determining the total magnitude of the three-phase quantity.
537    ///
538    /// # Parameters
539    ///
540    /// * `result` - Clarke transformation result
541    ///
542    /// # Returns
543    ///
544    /// The magnitude of the αβ space vector.
545    ///
546    /// # Examples
547    ///
548    /// ```rust
549    /// use libpower::transform::clarke::Clarke;
550    ///
551    /// let clarke = Clarke::new(0.0, 0.0);
552    /// let result = clarke.calculate(100.0, -50.0, -50.0);
553    /// let magnitude = clarke.magnitude(&result);
554    ///
555    /// assert!(magnitude > 0.0);
556    /// println!("Space vector magnitude: {:.2}", magnitude);
557    /// ```
558    pub fn magnitude(&self, result: &ClarkeResult) -> f32 {
559        sqrtf(result.alpha * result.alpha + result.beta * result.beta)
560    }

562    /// Calculates the phase angle of the αβ vector.
563    ///
564    /// Computes the angle of the space vector in the αβ plane using atan2.
565    /// The angle is measured from the α-axis towards the β-axis.
566    ///
567    /// # Parameters
568    ///
569    /// * `result` - Clarke transformation result
570    ///
571    /// # Returns
572    ///
573    /// The phase angle in radians [-π, π].
574    ///
575    /// # Examples
576    ///
577    /// ```rust
578    /// use libpower::transform::clarke::Clarke;
579    /// use std::f32::consts::PI;
580    ///
581    /// let clarke = Clarke::new(0.0, 0.0);
582    /// let result = clarke.calculate(100.0, -50.0, -50.0);
583    /// let angle = clarke.angle(&result);
584    ///
585    /// // Angle should be between -π and π
586    /// assert!(angle >= -PI && angle <= PI);
587    /// println!("Space vector angle: {:.2} rad ({:.1}°)",
588    ///          angle, angle * 180.0 / PI);
589    /// ```
590    pub fn angle(&self, result: &ClarkeResult) -> f32 {
591        atan2f(result.beta, result.alpha)
592    }

594    /// Checks if the three-phase system is balanced.
595    ///
596    /// A balanced three-phase system should have a zero-sequence component
597    /// close to zero. This method checks if the zero-sequence is within
598    /// a specified tolerance.
599    ///
600    /// # Parameters
601    ///
602    /// * `result` - Clarke transformation result
603    /// * `tolerance` - Maximum allowed zero-sequence magnitude for balanced system
604    ///
605    /// # Returns
606    ///
607    /// `true` if the system is considered balanced, `false` otherwise.
608    ///
609    /// # Examples
610    ///
611    /// ```rust
612    /// use libpower::transform::clarke::Clarke;
613    ///
614    /// let clarke = Clarke::new(0.0, 0.0);
615    ///
616    /// // Balanced system
617    /// let balanced = clarke.calculate(100.0, -50.0, -50.0);
618    /// assert!(clarke.is_balanced(&balanced, 1.0));
619    ///
620    /// // Unbalanced system  
621    /// let unbalanced = clarke.calculate(120.0, -50.0, -50.0);
622    /// assert!(!clarke.is_balanced(&unbalanced, 1.0));
623    /// ```
624    pub fn is_balanced(&self, result: &ClarkeResult, tolerance: f32) -> bool {
625        result.zero.abs() <= tolerance
626    }
