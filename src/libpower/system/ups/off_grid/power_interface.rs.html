1/// Power-analyzer–grade abstraction for an electrical power interface.
2///
3/// Suitable for: mains input, inverter output, battery port,
4/// solar input, DC loads, bidirectional converters.
5pub trait PowerInterface {
6    /// ---------- Basic electrical ----------
7
8    /// Voltage (RMS for AC, DC value for DC)
9    fn voltage(&self) -> f32;
10
11    /// Current (RMS for AC, DC value for DC)
12    /// Positive = power into the system
13    fn current(&self) -> f32;
14
15    /// Frequency in Hz (None for DC)
16    fn frequency(&self) -> Option<f32>;
17
18    /// Phase angle between voltage and current (degrees, AC only)
19    fn phase_angle(&self) -> Option<f32>;
20
21    /// ---------- Power quantities ----------
22
23    /// Active (real) power P (W)
24    fn active_power(&self) -> f32;
25
26    /// Reactive power Q (VAR, AC only)
27    fn reactive_power(&self) -> Option<f32>;
28
29    /// Apparent power S (VA, AC only)
30    fn apparent_power(&self) -> Option<f32>;
31
32    /// Power factor (true PF, includes harmonics)
33    fn power_factor(&self) -> Option<f32>;
34
35    /// Displacement power factor (fundamental only)
36    fn displacement_pf(&self) -> Option<f32>;
37
38    /// ---------- Energy ----------
39
40    /// Energy imported into the system (Wh)
41    fn energy_imported(&self) -> f64;
42
43    /// Energy exported from the system (Wh)
44    fn energy_exported(&self) -> f64;
45
46    /// ---------- Power quality ----------
47
48    /// Voltage THD (%)
49    fn voltage_thd(&self) -> Option<f32>;
50
51    /// Current THD (%)
52    fn current_thd(&self) -> Option<f32>;
53
54    /// Crest factor of voltage
55    fn voltage_crest_factor(&self) -> Option<f32>;
56
57    /// Crest factor of current
58    fn current_crest_factor(&self) -> Option<f32>;
59
60    /// Individual harmonic magnitude (n = 1,2,3...)
61    /// Returned as percentage of fundamental
62    fn voltage_harmonic(&self, n: u8) -> Option<f32>;
63
64    fn current_harmonic(&self, n: u8) -> Option<f32>;
65
66    /// ---------- Efficiency & losses ----------
67
68    /// Conversion efficiency (%) if applicable
69    fn efficiency(&self) -> Option<f32>;
70
71    /// Estimated losses (W), if known
72    fn losses(&self) -> Option<f32>;
73
74    /// ---------- Limits & ratings ----------
75
76    fn voltage_min(&self) -> f32;
77    fn voltage_max(&self) -> f32;
78    fn current_max(&self) -> f32;
79    fn power_max(&self) -> f32;
80
81    /// ---------- State & health ----------
82
83    fn is_enabled(&self) -> bool;
84    fn is_faulted(&self) -> bool;
85
86    /// Interface temperature (°C), if available
87    fn temperature(&self) -> Option<f32>;
88
89    /// ---------- Convenience defaults ----------
90
91    /// Instantaneous power derived when analyzer doesn’t provide P directly
92    fn instantaneous_power(&self) -> f32 {
93        self.voltage() * self.current()
94    }
95}
