1/// Minimal serial communication trait and mock implementation for UPS physical components.
2///
3/// This module provides a basic serial interface inspired by USART drivers,
4/// with send and receive operations for byte-level communication.
5
6/// A generic error type for serial operations, using i32 for error codes.
7pub type Error = i32;
8
9/// A specialized Result type for serial operations.
10pub type Result<T> = core::result::Result<T, Error>;
11
12/// Minimal trait for serial communication operations.
13pub trait Serial {
14    /// Send data over the serial interface.
15    fn send(&mut self, data: &[u8]) -> Result<()>;
16
17    /// Receive data from the serial interface into the provided buffer.
18    /// Returns the number of bytes received.
19    fn receive(&mut self, buffer: &mut [u8]) -> Result<usize>;
20}
21
22/// Mock implementation of the Serial trait for testing and default behavior.
23#[derive(Debug, Clone, Default)]
24pub struct SerialMock;
25
26impl Serial for SerialMock {
27    fn send(&mut self, _data: &[u8]) -> Result<()> {
28        // Mock implementation: always succeed
29        Ok(())
30    }
31
32    fn receive(&mut self, buffer: &mut [u8]) -> Result<usize> {
33        // Mock implementation: no data received
34        for byte in buffer.iter_mut() {
35            *byte = 0;
36        }
37        Ok(0)
38    }
39}
