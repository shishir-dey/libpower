6/// Temperature monitoring service that monitors temperature sensors and controls a fan.
7///
8/// The service turns the fan on when the maximum temperature exceeds the threshold,
9/// and turns it off when the temperature falls below the threshold minus hysteresis.
10pub struct TemperatureMonitoringService<T1, T2, F>
11where
12    T1: TemperatureSensor,
13    T2: TemperatureSensor,
14    F: Fan,
15{
16    temp1: T1,
17    temp2: T2,
18    fan: F,
19    threshold_on: f32,
20    hysteresis: f32,
21}
22
23impl<T1, T2, F> TemperatureMonitoringService<T1, T2, F>
24where
25    T1: TemperatureSensor,
26    T2: TemperatureSensor,
27    F: Fan,
28{
29    /// Create a new temperature monitoring service.
30    ///
31    /// # Arguments
32    ///
33    /// * `temp1` - First temperature sensor
34    /// * `temp2` - Second temperature sensor
35    /// * `fan` - Fan controller
36    /// * `threshold_on` - Temperature threshold to turn fan on (in Celsius)
37    /// * `hysteresis` - Hysteresis value for turning fan off (in Celsius)
38    pub fn new(temp1: T1, temp2: T2, fan: F, threshold_on: f32, hysteresis: f32) -> Self {
39        Self {
40            temp1,
41            temp2,
42            fan,
43            threshold_on,
44            hysteresis,
45        }
46    }
47
48    /// Update the temperature monitoring logic.
49    ///
50    /// Checks the maximum temperature from both sensors and controls the fan accordingly.
51    pub fn update(&mut self) {
52        let temp1 = self.temp1.get_temperature();
53        let temp2 = self.temp2.get_temperature();
54        let max_temp = temp1.max(temp2);
55        let threshold_off = self.threshold_on - self.hysteresis;
56
57        if max_temp > self.threshold_on {
58            self.fan.on();
59            self.fan.set_speed(1.0);
60        } else if max_temp < threshold_off {
61            self.fan.off();
62        }
63        // If between threshold_off and threshold_on, maintain current fan state
64    }
65}
66
67impl<T1, T2, F> Service for TemperatureMonitoringService<T1, T2, F>
68where
69    T1: TemperatureSensor,
70    T2: TemperatureSensor,
71    F: Fan,
72{
73    fn update(&mut self) {
74        self.update();
