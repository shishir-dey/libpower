1use crate::system::ups::off_grid::physical::serial::Serial;
2/// Modbus master implementation for UPS system services.
3///
4/// This module provides a Modbus RTU master for communicating with devices
5/// like MPPT controllers, using the serial interface.
6use alloc::boxed::Box;
7use heapless::Vec;
8
9const MODBUS_SLAVE_ID: u8 = 1;
10const MODBUS_FUNCTION_READ_HOLDING_REGISTERS: u8 = 0x03;
11const MODBUS_CRC_POLYNOMIAL: u16 = 0xA001;
12
13/// Modbus communication states.
14#[derive(Debug, Clone, Copy, PartialEq, Eq)]
15pub enum ModbusState {
16    Idle,
17    Transmitting,
18    Receiving,
19    Processing,
20}
21
22/// Modbus request structure.
23#[derive(Debug, Clone)]
24pub struct ModbusRequest {
25    pub slave_id: u8,
26    pub function_code: u8,
27    pub start_address: u16,
28    pub num_registers: u16,
29    pub tx_buffer: [u8; 8],
30    pub rx_buffer: Vec<u8, 256>,
31    pub state: ModbusState,
32    pub crc: u16,
33}
34
35impl Default for ModbusRequest {
36    fn default() -> Self {
37        Self {
38            slave_id: 0,
39            function_code: 0,
40            start_address: 0,
41            num_registers: 0,
42            tx_buffer: [0; 8],
43            rx_buffer: Vec::new(),
44            state: ModbusState::Idle,
45            crc: 0,
46        }
47    }
48}
49
50/// Modbus master structure.
51pub struct ModbusMaster {
52    serial: Box<dyn Serial>,
53    current_request: ModbusRequest,
54}
55
56impl ModbusMaster {
57    /// Create a new Modbus master with the given serial interface.
58    pub fn new(serial: Box<dyn Serial>) -> Self {
59        Self {
60            serial,
61            current_request: ModbusRequest::default(),
62        }
63    }
64
65    /// Initialize the Modbus master.
66    pub fn init(&mut self) {
67        self.current_request.state = ModbusState::Idle;
68    }
69
70    /// Calculate Modbus CRC16.
71    fn modbus_crc16(data: &[u8]) -> u16 {
72        let mut crc = 0xFFFFu16;
73        for &byte in data {
74            crc ^= byte as u16;
75            for _ in 0..8 {
76                if crc & 1 != 0 {
77                    crc = (crc >> 1) ^ MODBUS_CRC_POLYNOMIAL;
78                } else {
79                    crc >>= 1;
80                }
81            }
82        }
83        crc
84    }
85
86    /// Read holding registers from a Modbus slave.
87    pub fn read_holding_registers(
88        &mut self,
89        slave_id: u8,
90        start_address: u16,
91        num_registers: u16,
92    ) -> bool {
93        if self.current_request.state != ModbusState::Idle {
94            return false;
95        }
96
97        self.current_request.slave_id = slave_id;
98        self.current_request.function_code = MODBUS_FUNCTION_READ_HOLDING_REGISTERS;
99        self.current_request.start_address = start_address;
100        self.current_request.num_registers = num_registers;
101        self.current_request.state = ModbusState::Transmitting;
102        self.current_request.rx_buffer.clear();
103
104        // Build request frame
105        let mut tx_buffer = [0u8; 8];
106        tx_buffer[0] = slave_id;
107        tx_buffer[1] = MODBUS_FUNCTION_READ_HOLDING_REGISTERS;
108        tx_buffer[2] = (start_address >> 8) as u8;
109        tx_buffer[3] = start_address as u8;
110        tx_buffer[4] = (num_registers >> 8) as u8;
111        tx_buffer[5] = num_registers as u8;
112
113        let crc = Self::modbus_crc16(&tx_buffer[..6]);
114        tx_buffer[6] = crc as u8;
115        tx_buffer[7] = (crc >> 8) as u8;
116
117        self.current_request.tx_buffer = tx_buffer;
118
119        // Send
120        if self.serial.send(&tx_buffer).is_ok() {
121            self.current_request.state = ModbusState::Receiving;
122            true
123        } else {
124            self.current_request.state = ModbusState::Idle;
125            false
126        }
127    }
128
129    /// Process incoming data and handle state transitions.
130    pub fn process(&mut self) {
131        if self.current_request.state == ModbusState::Receiving {
132            // Try to receive data
133            let mut temp_buffer = [0u8; 256];
134            match self.serial.receive(&mut temp_buffer) {
135                Ok(len) => {
136                    for &byte in &temp_buffer[..len] {
137                        if self.current_request.rx_buffer.len() < 256 {
138                            self.current_request.rx_buffer.push(byte).unwrap();
139                        }
140                    }
141                    // For now, just read some bytes and do nothing
142                    self.current_request.state = ModbusState::Idle;
143                }
144                Err(_) => {
145                    // Error handling (e.g., timeout)
146                }
147            }
148        }
149    }
150
151    /// Take the received data.
152    pub fn take_rx_data(&mut self) -> Option<Vec<u8, 256>> {
153        if self.current_request.state == ModbusState::Idle
154            && !self.current_request.rx_buffer.is_empty()
155        {
156            let data = self.current_request.rx_buffer.clone();
157            self.current_request.rx_buffer.clear();
158            Some(data)
159        } else {
160            None
161        }
162    }
163
164    /// Get the current request state.
165    pub fn get_current_request(&self) -> &ModbusRequest {
166        &self.current_request
167    }
168}
