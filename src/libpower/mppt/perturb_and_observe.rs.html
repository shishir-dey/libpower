1//! # Perturb and Observe (P&O) MPPT Algorithm
2//!
3//! This module implements the classic Perturb and Observe Maximum Power Point Tracking
4//! algorithm commonly used in photovoltaic systems. The P&O algorithm is based on the
5//! hill-climbing principle where the operating point is perturbed and the resulting
6//! power change is observed to determine the direction toward the maximum power point.
7//!
8//! ## Theory
9//!
10//! The P&O algorithm operates on the principle that:
11//! - At the MPP: dP/dV = 0 (slope of P-V curve is zero)
12//! - Left of MPP: dP/dV > 0 (increasing voltage increases power)
13//! - Right of MPP: dP/dV < 0 (increasing voltage decreases power)
14//!
15//! ## Algorithm Logic
16//!
17//! ```text
18//! 1. Measure current PV voltage and current
19//! 2. Calculate current power: P(k) = V(k) × I(k)
20//! 3. Compare with previous power: ΔP = P(k) - P(k-1)
21//! 4. Determine perturbation direction:
22//!    - If ΔP > 0: Continue in same direction
23//!    - If ΔP < 0: Reverse direction
24//! 5. Apply voltage perturbation: V_ref = V_ref ± step_size
25//! 6. Update previous values for next iteration
26//! ```
27//!
28//! ## Advantages
29//!
30//! - **Simplicity**: Easy to implement and understand
31//! - **Low computational cost**: Minimal processing requirements
32//! - **Hardware independence**: Works with any converter topology
33//! - **No PV array parameters required**: Model-free operation
34//!
35//! ## Limitations
36//!
37//! - **Steady-state oscillation**: Continuously oscillates around MPP
38//! - **Poor dynamic response**: Can be confused by rapid irradiance changes
39//! - **Drift under uniform irradiance**: May drift away from true MPP
40//! - **Step size trade-off**: Large steps = fast tracking but more oscillation
41//!
42//! ## Applications
43//!
44//! - Grid-tied solar inverters
45//! - Battery charging systems  
46//! - Standalone PV systems
47//! - Water pumping systems
48//! - Small-scale solar installations
49
50/// MPPT action enumeration for internal state tracking.
51///
52/// This enum represents the direction of the next voltage perturbation
53/// based on the observed power change from the previous iteration.
54#[derive(Debug, Clone, Copy)]
55enum VMPPAction {
56    /// Increase the reference voltage
57    INCREMENT,
58    /// Decrease the reference voltage  
59    DECREMENT,
60}

62/// Perturb and Observe MPPT controller implementation.
63///
64/// This struct maintains all the state variables and parameters required for
65/// the P&O MPPT algorithm operation, including measurement storage, power
66/// calculation, and voltage reference generation.
67///
68/// # Examples
69///
70/// ## Basic Usage
71///
72/// ```rust
73/// use libpower::mppt::perturb_and_observe::MPPT;
74///
75/// // Create and configure MPPT controller
76/// let mut mppt = MPPT::new();
77/// mppt.set_step_size(0.1);        // 0.1V voltage steps
78/// mppt.set_mppt_v_out_max(30.0);  // 30V maximum output
79/// mppt.set_mppt_v_out_min(12.0);  // 12V minimum output
80///
81/// // MPPT control loop (called periodically, e.g., every 100ms)
82/// let pv_current = 5.2;   // Measured PV current in Amperes
83/// let pv_voltage = 18.5;  // Measured PV voltage in Volts
84/// mppt.calculate(pv_current, pv_voltage);
85///
86/// let voltage_reference = mppt.get_mppt_v_out();
87/// println!("Set converter reference to: {:.2}V", voltage_reference);
88/// ```
89///
90/// ## Complete System Integration
91///
92/// ```rust
93/// use libpower::mppt::perturb_and_observe::MPPT;
94///
95/// let mut mppt = MPPT::new();
96/// mppt.set_step_size(0.05);       // Small steps for fine tracking
97/// mppt.set_mppt_v_out_max(24.0);  // Battery voltage limit
98/// mppt.set_mppt_v_out_min(10.0);  // Minimum operating voltage
99///
100/// // Simulate several control cycles
101/// let measurements = [
102///     (4.8, 17.2),  // (Current, Voltage) measurements
103///     (4.9, 17.4),  // Power increasing: 82.56W → 85.26W
104///     (5.0, 17.3),  // Power slightly decreasing: 85.26W → 86.5W
105///     (5.1, 17.1),  // Continue tracking...
106/// ];
107///
108/// for (i_pv, v_pv) in measurements.iter() {
109///     mppt.calculate(*i_pv, *v_pv);
110///     let v_ref = mppt.get_mppt_v_out();
111///     let power = mppt.get_pv_power();
112///     println!("Cycle: P={:.1}W, V_ref={:.2}V", power, v_ref);
113/// }
114/// ```
115pub struct MPPT {
116    /// Current PV current measurement (A)
117    pv_i: f32,
118    /// Current PV voltage measurement (V)
119    pv_v: f32,
120    /// Previous PV voltage measurement (V)
121    pv_v_prev: f32,
122    /// Current PV power calculation (W)
123    pv_power: f32,
124    /// Previous PV power calculation (W)
125    pv_power_prev: f32,
126    /// Power change from previous iteration (W)
127    delta_pv_power: f32,
128    /// Minimum power change threshold (W) - currently unused
129    delta_p_min: f32,
130    /// Next MPPT action (INCREMENT or DECREMENT)
131    mppt_v_out_action: VMPPAction,
132    /// Maximum voltage reference limit (V)
133    mppt_v_out_max: f32,
134    /// Minimum voltage reference limit (V)
135    mppt_v_out_min: f32,
136    /// Voltage step size for perturbation (V)
137    step_size: f32,
138    /// Current voltage reference output (V)
139    mppt_v_out: f32,
140    /// MPPT enable flag (currently unused)
141    mppt_enable: bool,
142    /// First calculation flag to skip initial comparison
143    mppt_first: bool,
144}

337    /// Executes one iteration of the P&O MPPT algorithm.
338    ///
339    /// This is the main method that implements the perturb and observe logic.
340    /// It should be called periodically (typically every 10-1000ms) with fresh
341    /// PV measurements to track the maximum power point.
342    ///
343    /// # Parameters
344    ///
345    /// * `pv_i` - Current PV current measurement in Amperes
346    /// * `pv_v` - Current PV voltage measurement in Volts
347    ///
348    /// # Algorithm Steps
349    ///
350    /// 1. Store previous measurements (first call only)
351    /// 2. Calculate current power: P = V × I
352    /// 3. Compare with previous power: ΔP = P(k) - P(k-1)
353    /// 4. Determine voltage perturbation direction based on ΔP and ΔV signs
354    /// 5. Apply voltage step and limit to configured bounds
355    /// 6. Update previous values for next iteration
356    ///
357    /// # Examples
358    ///
359    /// ## Basic Operation
360    ///
361    /// ```rust
362    /// use libpower::mppt::perturb_and_observe::MPPT;
363    ///
364    /// let mut mppt = MPPT::new();
365    /// mppt.set_step_size(0.1);
366    /// mppt.set_mppt_v_out_max(25.0);
367    /// mppt.set_mppt_v_out_min(10.0);
368    ///
369    /// // Call periodically with sensor measurements
370    /// mppt.calculate(5.0, 18.0);  // First call - just stores values
371    /// mppt.calculate(5.1, 18.2);  // Second call - starts tracking
372    ///
373    /// let v_ref = mppt.get_mppt_v_out();  // Use this for converter control
374    /// ```
375    ///
376    /// ## Simulated Tracking
377    ///
378    /// ```rust
379    /// use libpower::mppt::perturb_and_observe::MPPT;
380    ///
381    /// let mut mppt = MPPT::new();
382    /// mppt.set_step_size(0.05);
383    /// mppt.set_mppt_v_out_max(20.0);
384    /// mppt.set_mppt_v_out_min(15.0);
385    ///
386    /// // Simulate approaching MPP
387    /// let test_points = [
388    ///     (4.0, 16.0),  // 64W - below MPP
389    ///     (4.5, 17.0),  // 76.5W - approaching MPP  
390    ///     (5.0, 17.5),  // 87.5W - closer to MPP
391    ///     (4.8, 17.8),  // 85.44W - past MPP, power decreasing
392    /// ];
393    ///
394    /// for (current, voltage) in test_points.iter() {
395    ///     mppt.calculate(*current, *voltage);
396    ///     println!("Power: {:.1}W, V_ref: {:.2}V",
397    ///              mppt.get_pv_power(), mppt.get_mppt_v_out());
398    /// }
399    /// ```
400    pub fn calculate(&mut self, pv_i: f32, pv_v: f32) {
401        if self.mppt_first {
402            // Store current measurements for both old and current values on first call
403            self.pv_i = pv_i;
404            self.pv_v = pv_v;
405            self.pv_power = pv_i * pv_v;
406            self.pv_v_prev = pv_v;
407            self.pv_power_prev = self.pv_power;
408            self.mppt_first = false;
409        } else {
410            self.pv_i = pv_i;
411            self.pv_v = pv_v;
412            self.pv_power = self.pv_i * self.pv_v;
413
414            // Calculate power change
415            let delta_pv_power = self.pv_power - self.pv_power_prev;
416            if delta_pv_power > self.delta_p_min {
417                // Determine whether to INCREMENT or DECREMENT
418                if self.pv_v > self.pv_v_prev {
419                    self.mppt_v_out_action = VMPPAction::INCREMENT;
420                } else {
421                    self.mppt_v_out_action = VMPPAction::DECREMENT;
422                }
423
424                // Adjust voltage output based on action
425                match self.mppt_v_out_action {
426                    VMPPAction::INCREMENT => {
427                        if self.mppt_v_out + self.step_size > self.mppt_v_out_max {
428                            self.mppt_v_out = self.mppt_v_out_max;
429                        } else {
430                            self.mppt_v_out += self.step_size;
431                        }
432                    }
433                    VMPPAction::DECREMENT => {
434                        if self.mppt_v_out - self.step_size < self.mppt_v_out_min {
435                            self.mppt_v_out = self.mppt_v_out_min;
436                        } else {
437                            self.mppt_v_out -= self.step_size;
438                        }
439                    }
440                }
441            }
442            // Save the previous values
443            self.pv_v_prev = self.pv_v;
444            self.pv_power_prev = self.pv_power;
445        }
446    }
